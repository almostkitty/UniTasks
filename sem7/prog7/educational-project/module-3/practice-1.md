from docx import Document
from docx.shared import Pt, Inches
from docx.enum.text import WD_PARAGRAPH_ALIGNMENT

class TestGenerator:
    """–ì–µ–Ω–µ—Ä–∞—Ç–æ—Ä –∫–æ–Ω—Ç—Ä–æ–ª—å–Ω—ã—Ö —Ä–∞–±–æ—Ç"""
    
    def __init__(self, subject="–ú–∞—Ç–µ–º–∞—Ç–∏–∫–∞", class_name="9–ê"):
        self.subject = subject
        self.class_name = class_name
        self.variants_log = []
    
    def generate_quadratic_equation(self):
        """–ì–µ–Ω–µ—Ä–∞—Ü–∏—è –∫–≤–∞–¥—Ä–∞—Ç–Ω–æ–≥–æ —É—Ä–∞–≤–Ω–µ–Ω–∏—è"""
        a = random.randint(1, 5)
        b = random.randint(-10, 10)
        c = random.randint(-10, 10)
        
        # –í—ã—á–∏—Å–ª–µ–Ω–∏–µ –∫–æ—Ä–Ω–µ–π –¥–ª—è –æ—Ç–≤–µ—Ç–∞
        discriminant = b**2 - 4*a*c
        
        if discriminant >= 0:
            x1 = (-b + discriminant**0.5) / (2*a)
            x2 = (-b - discriminant**0.5) / (2*a)
            answer = f"x‚ÇÅ = {x1:.2f}, x‚ÇÇ = {x2:.2f}"
        else:
            answer = "–ù–µ—Ç –¥–µ–π—Å—Ç–≤–∏—Ç–µ–ª—å–Ω—ã—Ö –∫–æ—Ä–Ω–µ–π"
        
        task = {
            'question': f"–†–µ—à–∏—Ç–µ —É—Ä–∞–≤–Ω–µ–Ω–∏–µ: {a}x¬≤ + {b}x + {c} = 0",
            'answer': answer,
            'points': 3
        }
        return task
    
    def generate_arithmetic_task(self):
        """–ì–µ–Ω–µ—Ä–∞—Ü–∏—è –∞—Ä–∏—Ñ–º–µ—Ç–∏—á–µ—Å–∫–æ–π –∑–∞–¥–∞—á–∏"""
        num1 = random.randint(10, 100)
        num2 = random.randint(10, 100)
        operations = ['+', '-', '*']
        op = random.choice(operations)
        
        if op == '+':
            answer = num1 + num2
        elif op == '-':
            answer = num1 - num2
        else:
            answer = num1 * num2
        
        task = {
            'question': f"–í—ã—á–∏—Å–ª–∏—Ç–µ: {num1} {op} {num2}",
            'answer': str(answer),
            'points': 1
        }
        return task
    
    def generate_word_problem(self):
        """–ì–µ–Ω–µ—Ä–∞—Ü–∏—è —Ç–µ–∫—Å—Ç–æ–≤–æ–π –∑–∞–¥–∞—á–∏"""
        scenarios = [
            {
                'template': '–¢–æ–≤–∞—Ä —Å—Ç–æ–∏–ª {price} —Ä—É–±. –ü–æ—Å–ª–µ —Å–∫–∏–¥–∫–∏ {discount}% –µ–≥–æ —Ü–µ–Ω–∞ —Å–æ—Å—Ç–∞–≤–∏–ª–∞ {final} —Ä—É–±. –ö–∞–∫–æ–≤–∞ –±—ã–ª–∞ —Å–∫–∏–¥–∫–∞ –≤ —Ä—É–±–ª—è—Ö?',
                'price': random.randint(1000, 5000),
                'discount': random.randint(10, 30)
            },
            {
                'template': '–ê–≤—Ç–æ–º–æ–±–∏–ª—å –ø—Ä–æ–µ—Ö–∞–ª {distance} –∫–º –∑–∞ {time} —á–∞—Å–æ–≤. –ö–∞–∫–æ–≤–∞ —Å—Ä–µ–¥–Ω—è—è —Å–∫–æ—Ä–æ—Å—Ç—å?',
                'distance': random.randint(100, 500),
                'time': random.randint(2, 8)
            }
        ]
        
        scenario = random.choice(scenarios)
        
        if 'price' in scenario:
            price = scenario['price']
            discount = scenario['discount']
            final = price * (1 - discount/100)
            discount_amount = price - final
            question = scenario['template'].format(price=price, discount=discount, final=int(final))
            answer = f"{discount_amount:.0f} —Ä—É–±."
        else:
            distance = scenario['distance']
            time = scenario['time']
            speed = distance / time
            question = scenario['template'].format(distance=distance, time=time)
            answer = f"{speed:.1f} –∫–º/—á"
        
        task = {
            'question': question,
            'answer': answer,
            'points': 2
        }
        return task
    
    def generate_test_variant(self, variant_number, num_tasks=10):
        """–ì–µ–Ω–µ—Ä–∞—Ü–∏—è –≤–∞—Ä–∏–∞–Ω—Ç–∞ —Ç–µ—Å—Ç–∞"""
        tasks = []
        
        # 40% –∫–≤–∞–¥—Ä–∞—Ç–Ω—ã–µ —É—Ä–∞–≤–Ω–µ–Ω–∏—è
        for _ in range(int(num_tasks * 0.4)):
            tasks.append(self.generate_quadratic_equation())
        
        # 30% –∞—Ä–∏—Ñ–º–µ—Ç–∏–∫–∞
        for _ in range(int(num_tasks * 0.3)):
            tasks.append(self.generate_arithmetic_task())
        
        # 30% —Ç–µ–∫—Å—Ç–æ–≤—ã–µ –∑–∞–¥–∞—á–∏
        for _ in range(num_tasks - len(tasks)):
            tasks.append(self.generate_word_problem())
        
        random.shuffle(tasks)
        
        return {
            'variant': variant_number,
            'tasks': tasks,
            'total_points': sum(t['points'] for t in tasks),
            'date': datetime.now().strftime('%d.%m.%Y')
        }
    
    def create_word_document(self, variant_data, filename):
        """–°–æ–∑–¥–∞–Ω–∏–µ Word –¥–æ–∫—É–º–µ–Ω—Ç–∞ —Å –∑–∞–¥–∞–Ω–∏—è–º–∏"""

        info.add_run(f'–í–∞—Ä–∏–∞–Ω—Ç: {variant_data["variant"]}\n').bold = True
        info.add_run(f'–î–∞—Ç–∞: {variant_data["date"]}\n')
        info.add_run(f'–ú–∞–∫—Å–∏–º–∞–ª—å–Ω—ã–π –±–∞–ª–ª: {variant_data["total_points"]}\n')
        
        doc.add_paragraph('–§–∞–º–∏–ª–∏—è, –ò–º—è: _' + '_' * 40)
        doc.add_paragraph('-' * 60)
        
        # –ó–∞–¥–∞–Ω–∏—è
        for i, task in enumerate(variant_data['tasks'], 1):
            doc.add_heading(f'–ó–∞–¥–∞–Ω–∏–µ {i} ({task["points"]} –±–∞–ª–ª.)', level=2)

        # –ß–µ—Ä–Ω–æ–≤–∏–∫
        doc.add_page_break()

        doc.save(filename)

        doc.add_paragraph(f'–î–∞—Ç–∞: {datetime.now().strftime("%d.%m.%Y")}')
